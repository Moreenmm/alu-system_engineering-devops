#!/usr/bin/env bash # Script that installs nginx and configures a creative 404 page sudo apt update -y sudo apt install nginx -y # Create a beautiful custom 404 page cat << 'EOF' > /var/www/html/custom_404.html 404 Not Found

404

Ceci n'est pas une page

Oops! You seem lost in cyberspace.

Go Back Home EOF # Configure nginx to use the custom 404 page sed -i '/server_name _;/a \ error_page 404 /custom_404.html;\n location = /custom_404.html {\n root /var/www/html;\n internal;\n }' /etc/nginx/sites-available/default # Restart nginx sudo service nginx restart
